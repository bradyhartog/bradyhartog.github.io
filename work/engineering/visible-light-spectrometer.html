Demonstrating the operation of the visible light spectrometer.

// Initialize input pins

const int red_input = A0;
const int green_input = A1;
const int blue_input = A2;

// Initialize output pins

const int red_indicator_low = 2;
const int red_indicator_mid = 3;
const int red_indicator_high = 4;

const int green_indicator_low = 5;
const int green_indicator_mid = 6;
const int green_indicator_high = 7;

const int blue_indicator_low = 8;
const int blue_indicator_mid = 9;
const int blue_indicator_high = 10;

// Initialize channel values

int R = 0;
int G = 0;
int B = 0;

void setup() {
  Serial.begin(9600);

  pinMode(red_indicator_low, OUTPUT);
  pinMode(red_indicator_mid, OUTPUT);
  pinMode(red_indicator_high, OUTPUT);
  
  pinMode(green_indicator_low, OUTPUT);
  pinMode(green_indicator_mid, OUTPUT);
  pinMode(green_indicator_high, OUTPUT);
  
  pinMode(blue_indicator_low, OUTPUT);
  pinMode(blue_indicator_mid, OUTPUT);
  pinMode(blue_indicator_high, OUTPUT);
}

void loop() {
  // Read channel values
  
  R = analogRead(red_input);
  delay(5);
  G = analogRead(green_input);
  delay(5);
  B = analogRead(blue_input);

  // Print channel data

  Serial.print("Raw Sensor Values \t R: ");
  Serial.print(R);
  Serial.print("\t G: ");
  Serial.print(G);
  Serial.print("\t B: ");
  Serial.println(B);

  if (R >= 256) {
    digitalWrite(red_indicator_low, 1);
  } else {
    digitalWrite(red_indicator_low, 0);
  }
  if (R >= 512) {
    digitalWrite(red_indicator_mid, 1);
  } else {
    digitalWrite(red_indicator_mid, 0);
  }
  if (R >= 768) {
    digitalWrite(red_indicator_high, 1);
  } else {
    digitalWrite(red_indicator_high, 0);
  }

  if (G >= 256) {
    digitalWrite(green_indicator_low, 1);
  } else {
    digitalWrite(green_indicator_low, 0);
  }
  if (G >= 512) {
    digitalWrite(green_indicator_mid, 1);
  } else {
    digitalWrite(green_indicator_mid, 0);
  }
  if (G >= 768) {
    digitalWrite(green_indicator_high, 1);
  } else {
    digitalWrite(green_indicator_high, 0);
  }

  if (B >= 256) {
    digitalWrite(blue_indicator_low, 1);
  } else {
    digitalWrite(blue_indicator_low, 0);
  }
  if (B >= 512) {
    digitalWrite(blue_indicator_mid, 1);
  } else {
    digitalWrite(blue_indicator_mid, 0);
  }
  if (B >= 768) {
    digitalWrite(blue_indicator_high, 1);
  } else {
    digitalWrite(blue_indicator_high, 0);
  }
}
